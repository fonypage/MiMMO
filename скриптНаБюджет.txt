WHERE {
  :Remont_Kvartira1 :remontImeetOgranichenie :Byudzhet_Kvartira1 .
  :Remont_Kvartira1 :remontImeetSmetu ?smeta .

  :Byudzhet_Kvartira1 :byudzhetSumma ?budget .

  ?smeta :smetaImeetStoimostMaterialov ?smMat .
  ?smeta :smetaImeetStoimostRabot ?smRab .

  ?smMat :stoimostMaterialovSumma ?mat .
  ?smRab :stoimostRabotSumma ?work .
}

    direction LR

        [*] --> OffState
    }
    
    state Ready {
        [*] --> ReadyState
    }
    
    state Brewing {
        [*] --> BrewingState
    }
    
    state Error {
        [*] --> ErrorState
    }
    
    Off --> Ready: POWER (есть вода/зерна)
    Off --> Error: POWER (нет воды/зерен)
    Off --> Off: POWER 3 сек

    Brewing --> Ready: через 2 минуты
    Brewing --> Error: кончилась вода/зерна
    Brewing --> Off: POWER 3 сек
    
    Error --> Ready: добавили воду/зерна
    Error --> Off: POWER (любое)
