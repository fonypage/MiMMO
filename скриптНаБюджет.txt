SELECT ?budget ?mat ?work (?mat + ?work AS ?total) ((?mat + ?work) - ?budget AS ?over)
WHERE {
  :Remont_Kvartira1 :remontImeetOgranichenie :Byudzhet_Kvartira1 .
  :Remont_Kvartira1 :remontImeetSmetu ?smeta .

  :Byudzhet_Kvartira1 :byudzhetSumma ?budget .

  ?smeta :smetaImeetStoimostMaterialov ?smMat .
  ?smeta :smetaImeetStoimostRabot ?smRab .

  ?smMat :stoimostMaterialovSumma ?mat .
  ?smRab :stoimostRabotSumma ?work .
}
